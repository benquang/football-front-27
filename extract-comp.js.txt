for (let btn of document.getElementsByClassName('container-fluid modules')[0].getElementsByClassName('row')[0].getElementsByTagName('footer')) {
  let linkModule = btn.getElementsByTagName('a')[0].getAttribute('href').replace('./', 'http://localhost:8080/');
  re += await new Promise(resolve => {
    fetch(linkModule).then(res => res.text())
      .then(html => {
        let parser = new DOMParser();
        let doc = parser.parseFromString(html, "text/html");

        let a = `Module\t${doc.getElementsByClassName('breadcrumb')[0].getElementsByTagName('li')[1].getInnerHTML()}\t${defaultComp[doc.getElementsByClassName('breadcrumb')[0].getElementsByTagName('li')[1].getInnerHTML()] ?? ''}\t${linkModule}\n`;

        if (doc.getElementsByClassName('list-group').length) {
          for (let li of doc.getElementsByClassName('list-group')[0].getElementsByTagName('li')) {
            let compName = li.getElementsByTagName('a')[0].getInnerHTML();
            if (/.*Component$/.test(compName)) {
              a += `Component\t${li.getElementsByTagName('a')[0].getInnerHTML()}\t${defaultComp[li.getElementsByTagName('a')[0].getInnerHTML()] ?? ''}\t${li.getElementsByTagName('a')[0].getAttribute('href').replace('../', 'http://localhost:8080/')}\n`;
            }
          }
        }
        resolve(a);
      });
  });
}
